class KaggleFeatureExtractor:

        self.cae = QuantCAE()  # 你的模型
        self.cae.load_pretrained()  # 加载权重
    
    def extract_features(self, price_data):
        """将价格数据转为K线图，再提取特征"""
        # 1. 生成K线图
        chart_img = generate_candlestick(price_data)
        # 2. CAE编码
        features = self.cae.encode(chart_img)
        return features  # 1024维特征向量
